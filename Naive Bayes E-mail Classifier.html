def findW(wo,str_):
    ls = [len(x) for x in trainingSet[trainingSet.Category == str_].Text.str.findall(wo) if x != []]
    return sum(ls)
def findW1(wo,str_):
    ls = [len(x) for x in trainingSet[trainingSet.Category == str_].Text.str.findall(wo) if x != []]
    return len(ls) 
def process(row):
    k=0
    for w in row.Words:
        position = words2[words2.Words == w]["log(PWH/PWS)"]
        if len(position)>0:
            k+=position.values[0]
    if (k+phps) < 0:return "Spam"
    else:return "Ham"
    
def Confusion_Matrix(r):
    if r.Category == r.Label: return "True"+r.Label
    else: return "False"+r.Label

In [12]:
words["P(W|H)"] = words["HamMailsCount"]/len(trainingSet[trainingSet.Category == "Ham"])
words["P(W|S)"] = words["SpamMailsCount"]/len(trainingSet[trainingSet.Category == "Spam"])
#if value is 0, assign a small number in order to prevent calculation errors.
words.loc[words['P(W|H)'] == 0, 'P(W|H)'] = 0.000001
words.loc[words['P(W|S)'] == 0, 'P(W|S)'] = 0.000001
words["log(PWH/PWS)"] = np.log(words["P(W|H)"]/words["P(W|S)"])
phps = np.log(len(trainingSet[trainingSet.Category == "Ham"])/len(trainingSet[trainingSet.Category == "Spam"]))
